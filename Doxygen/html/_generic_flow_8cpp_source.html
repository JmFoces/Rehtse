58  /*
59  * Care with this. It will work only when first 8 bytes of the packet are the ports.

62  */
63  BOOST_LOG_TRIVIAL(debug) << "Updating with Generic Key";
64  Crafter::Layer *ip_overlay = (Crafter::Layer*) packet->GetTopLayer();
65  uint8_t *raw_data_ptr = (uint8_t *) malloc(ip_overlay->GetSize());
66  ip_overlay->GetData(raw_data_ptr);
67  uint16_t sport = raw_data_ptr[0];
68  uint16_t dport = raw_data_ptr[2];
